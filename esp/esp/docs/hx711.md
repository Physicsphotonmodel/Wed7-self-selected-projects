# How to initialize
Example：
```cpp
#include <Arduino.h>
#include "LoadCell.h"

// Define ESP32 safe pins
const uint8_t DT_PIN = 16;
const uint8_t SCK_PIN = 17;

// Instantiate the LoadCell object
LoadCell myScale(DT_PIN, SCK_PIN);

// Known weight for calibration (in your preferred unit, e.g., grams)
const float KNOWN_WEIGHT = 100.0; 

// Initial calibration factor (Replace with the calculated factor later)
float calibrationFactor = 1.0; 

void setup() {
    Serial.begin(115200);
    while (!Serial) { delay(10); }

    Serial.println("\n--- HX711 Load Cell Test ---");
    
    // Initialize sensor
    myScale.begin();
    
    // Reset scale to 0
    Serial.println("Taring... Please remove all weights.");
    delay(2000);
    myScale.tare();
    Serial.println("Tare completed.");

    // Put sample weight prompt
    Serial.println("Please place the known sample weight on the scale...");
    delay(5000); 
}

void loop() {
    // Read raw data (average of 10 readings)
    float rawValue = myScale.getWeight(10);
    
    // Calculate the calibration factor dynamically
    float currentFactor = rawValue / KNOWN_WEIGHT;
    
    Serial.print("Raw Value: ");
    Serial.print(rawValue);
    Serial.print(" | Calculated Scale Factor: ");
    Serial.println(currentFactor, 0); 
    
    delay(1000);
}
```
